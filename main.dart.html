55         textToShow = "$numberA - $numberB = $result"; 
56       });

60       int numberA = int.parse(controller_numberA.text); 
61       int numberB = int.parse(controller_numberB.text);

65       setState(() { 
66         textToShow = "$numberA / $numberB = $result";
